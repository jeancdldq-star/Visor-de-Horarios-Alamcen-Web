Limpieza de tiempos (Extremadamente robusta para evitar fallos si Excel usa fechas ocultas) def format_time(x): if pd.isna(x): return "" if hasattr(x, 'strftime'): return x.strftime("%H:%M") s = str(x).strip() if ' ' in s: s = s.split(' ')[-1] # Ignorar fecha si viene pegada if len(s) > 5: s = s[:5] if len(s) == 4 and s[1] == ':': s = "0" + s return s df['hora_inicio_limpia'] = df['hora inicio'].apply(format_time) df['hora_fin_limpia'] = df['hora fin'].apply(format_time) # Quitar filas inválidas vitales df = df[(df['hora_inicio_limpia'] != "") & (df['hora_fin_limpia'] != "")] df = df.dropna(subset=['nombre empleado']) # 2. Orden Cronológico Perfecto (Evitar que 00:00 se ordene antes que 22:00) def calcular_minutos_logicos(hora_str): try: h, m = map(int, hora_str.split(':')) h_logica = (h - 6) % 24 # El día empieza a las 06:00 return h_logica * 60 + m except: return 9999 df['minutos_orden'] = df['hora_inicio_limpia'].apply(calcular_minutos_logicos) df['nombre_upper'] = df['nombre empleado'].astype(str).str.strip().str.upper() # Ordenamos verdaderamente por cronología df = df.sort_values(by=['nombre_upper', 'minutos_orden']).reset_index(drop=True) # 3. Tratamiento de Tareas (Herencia de espacios en blanco) df['tarea_upper'] = df['sección tarea'].astype(str).str.strip().str.upper() # Cambiamos cualquier cosa vacía por None (nativo de Python para que pandas ffill funcione) df['tarea_upper'] = df['tarea_upper'].replace(['NAN', 'NAT', 'NULL', 'NONE', ''], None) # Rellenar tareas vacías con la tarea anterior de ese mismo operario df['tarea_upper'] = df.groupby('nombre_upper')['tarea_upper'].ffill() # Por si acaso la primera tarea del día es la que estaba vacía, rellenamos con la siguiente df['tarea_upper'] = df.groupby('nombre_upper')['tarea_upper'].bfill() df = df.dropna(subset=['tarea_upper']) # Extraemos la hora para la columna visual (ej: "12") df['hora_int'] = pd.to_numeric(df['hora_inicio_limpia'].str.split(':').str[0], errors='coerce') df = df.dropna(subset=['hora_int']) # 4. Creación de Bloques Continuos (Detectar turnos partidos) cambio_empleado = df['nombre_upper'] != df['nombre_upper'].shift(1) cambio_tarea = df['tarea_upper'] != df['tarea_upper'].shift(1) # --- CORRECCIÓN: Ignoramos los descansos para que un mismo operario en la misma tarea se agrupe --- df['bloque_id'] = (cambio_empleado | cambio_tarea).cumsum() # -------------------------------------------------------------------------------------------------- # 5. g['hora_fin_limpia'].iloc[-1], 'ultimo_idx': g.index[-1] }) ).reset_index() def obtener_siguiente_seccion(row): empleado = row['nombre_upper'] idx = row['ultimo_idx'] # Buscar en el df general qué
